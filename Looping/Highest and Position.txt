import java.util.Scanner;
import java.io.IOException;
 
public class Main {
 
    public static void main(String[] args) throws IOException {
 
        int highest = 0;
		int position = 0;
		int num;
		Scanner input = new Scanner(System.in);
		
		for (int i=1; i<101; i++) {
			num = input.nextInt();
			if (highest < num) {
				highest = num;
				position = i;
	}
   }
		System.out.println(highest);		
		System.out.println(position);	
 
    }
 
}
